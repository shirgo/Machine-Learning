1% GETMSWTFEAT Gets the Multiscale Wavelet Transform features, these
 2% include: Energy, Variance, Standard Deviation, and Waveform Length
 3% feat = getmswtfeat(x,winsize,wininc,SF)
 4% ------------------------------------------------------------------
 5% The signals in x are divided into multiple windows of size
 6% "winsize" and the windows are spaced "wininc" apart.
 7% Inputs
 8% ------
 9%    x: 		1D signal
10% Outputs
11% -------
12%    feat:     WT features organized as [Energy, Variance, Waveform Length, Entropy]
13
14% Example
15% -------
16% feat = getmswtfeat(rand(1024,1))
17% Assuming here rand(1024,1) (this can be any one dimensional signal,
18% for example EEG or EMG) is a one dimensional signal sampled at 32
19% for 32 seconds only. Utilizing a window size of 128 at 32 increments,
20% features are extracted from the wavelet tree.
21% I assumed 10 decomposition levels (J=10) below in the code.
22% For a full tree at 10 levels you should get 11 features
23% as we have decided to extract 4 types of features then we get 11 x 4 =44
24% features.

38function feat = getmswtfeat_new(x)
39[cfs,longs] = wavedec(x,10,'db4');
40level = length(longs)-2;
41
42dim = 1;
43if dim==1
44    cfs = cfs';
45    longs = longs';
46end
47
48numOfSIGs  = size(cfs,1);
49num_CFS_TOT = size(cfs,2);
50absCFS   = abs(cfs);
51cfs_POW2 = absCFS.^2;
52Energy  = sum(cfs_POW2,2);
53percentENER = 0*ones(size(cfs_POW2));
54notZER = (Energy>0);
55percentENER(notZER,:) = 100*cfs_POW2(notZER,:)./Energy(notZER,ones(1,num_CFS_TOT));
56
57%% Pre-define and allocate memory
58tab_ENER = zeros(numOfSIGs,level+1);
59tab_VAR = zeros(numOfSIGs,level+1);
60% tab_STD = zeros(numOfSIGs,level+1);
61tab_WL = zeros(numOfSIGs,level+1);
62tab_entropy = zeros(numOfSIGs,level+1);
63
64%% Feature extraction section
65st = 1;
66for k=1:level+1
67    nbCFS = longs(k);
68    en  = st+nbCFS-1;
69    tab_ENER(:,k) = mean(percentENER(:,st:en),2);%.*sum(abs(diff(absCFS0(:,st:en)')'),2); % energy per waveform length
70    tab_VAR(:,k) = var(percentENER(:,st:en),0,2); % variance of coefficients
71    %     tab_STD(:,k) = std(percentENER(:,st:en),[],2); % standard deviation of coefficients
72    tab_WL(:,k) = sum(abs(diff(percentENER(:,st:en)').^2))'; % waveform length
73    percentENER(:,st:en) = percentENER(:,st:en)./repmat(sum(percentENER(:,st:en),2),1,size(percentENER(:,st:en),2));
74    tab_entropy(:,k) = -sum(percentENER(:,st:en).*log(percentENER(:,st:en)),2)./size(percentENER(:,st:en),2);
75    st = en + 1;
76end
77feat =([log1p([tab_ENER tab_VAR  tab_WL]) tab_entropy]);
